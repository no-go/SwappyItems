426  std::set<Fid> candidates;
427  std::mutex m;
428  Fingerprint fp = getFingerprint(key);
429 
430  std::for_each (std::execution::par, _buckets.begin(), _buckets.end(), [&](Detail finfo) {
431  if ( finfo.minimum != finfo.maximum ) { // it is not deleted!
432 
433  if ( (key < finfo.minimum) || (key > finfo.maximum) ) {
434  // key is smaller then the smallest or bigger than the biggest
435  std::lock_guard lock(m);
436  ++statistic.rangeSaysNo;
437  } else {
438  // check bloom (is it possible, that this key is in that file?)
439  bool success = true;
440  for (auto b : fp) {
441  if (finfo.bloomMask[b] == false) {
442  success = false;
443  std::lock_guard lock(m);
444  ++statistic.bloomSaysNotIn;
445  break;
446  }
447  }
448  if (success) {
449  std::lock_guard lock(m);
450  candidates.insert(finfo.fid);
451  }
452  }
453  }
454  });
455 
456  keyFound = false;
457  done = 0;
458 
459  for (auto fid : candidates) {
460  if (!keyFound) {
461  std::thread th(&SwappyItems::loadFromFile, this, fid, key);
462  th.detach();
463  S->P();
464  }
465  }
466  if (candidates.size() > 0) while (done < candidates.size());
467 
468  if (keyFound) {
469  ++statistic.fileLoads;
470  return true;
471  } else {
472  if (candidates.size() > 0) {
473  // many candidates, but finaly not found in file
474  ++statistic.rangeFails;
475  }
