debian aus docker-hub installieren (weil image noch nicht lokal bekannt):
   docker run -it debian bash

   Workaround: mit "adduser dummy" einen user anlegen, passwd nicht vergessen,
   apt-get update, apt-get install vim mc xterm openssh-server aptitude
   contrib non-free zu main in /etc/apt/sources.list beifügen
   /etc/bash.bashrc: alias la="ls -lah --color=auto"
   /etc/vim/vimrc: syntax on, background=dark

mit dem ./Dockerfile ein image erzeugen:
   docker build -t imageBezeichnung .

alle images auflisten:
   docker images

images löschen:
   docker rmi imageBezeichnung
   docker rmi -f imageBezeichnung  (forciert löscht auch laufende container!)

welche container von welchem image laufen:
   docker ps
   docker ps -l (welche existieren?)

container löschen:
   docker rm containerId

einen dieser container (z.B. 72d468f455ea) als image speichern/überschreiben:
   docker commit 72d468f455ea benutzername/imageBezeichnung

einen container stoppen:
   docker stop 72d468f455ea
   
einen container wieder starten:
   docker start 72d468f455ea

eine bash in einem laufenden container starten:
   docker exec -ti 72d468f455ea bash

ein image starten und port 41088 auf den port 22 im Container umleiten:
   docker run -it -p 41088:22 benutzername/imageBezeichnung bash
     root@0e7c25e67738:/# /etc/init.d/ssh start
     [ ok ] Starting OpenBSD Secure Shell server: sshd.
     root@0e7c25e67738:/#

mit ssh auf container einloggen:
   benutzername@host> ssh -Y dummy@localhost -p 41088
     dummy@localhost's password: 
     dummy@0e7c25e67738:~$ xterm
     dummy@0e7c25e67738:~$ exit
